    TinyNOS — Справочник команд CLI¶

    Перечень команд настройки (set) и просмотра (show), сгруппированный по уровням сети — от физического порта до управления и эксплуатации, как в справочниках Nokia SR Linux, Arista EOS и Cisco NX-OS. Имена объектов (ACL, VRF, интерфейсы, peer-group) регистронезависимы. В примерах операционный режим обозначен промптом leaf1>, конфигурационный — leaf1#.

    Команды по уровням сети:

    • L1 — Физический уровень и порты
    • L2 — Канальный уровень
    • L3 — Сетевой уровень
    • Сетевая виртуализация (EVPN / VXLAN)
    • L4 — Сервисы, безопасность и QoS
    • Управление и эксплуатация
    • Справочно

    L1 — Физический уровень и порты¶

    Интерфейсы¶

    set interfaces <ifname> ipv4 address <cidr>                                       # Назначить IPv4 адрес/префикс на интерфейс
set interfaces <ifname> ipv6 address <cidr>                                       # Назначить IPv6 адрес/префикс на интерфейс
set interfaces <ifname> description <text>                                        # Текстовое описание порта (free-form, отображается в show)
set interfaces <ifname> mtu <mtu>                                                 # MTU L2-фрейма (64–9216, по умолчанию 1500)
set interfaces <ifname> admin-state up|down                                       # Административное состояние порта (up = поднять, down = погасить)
set interfaces <ifname> address <mac>                                             # Жёстко задать MAC-адрес (формат aa:bb:cc:dd:ee:ff)
set interfaces <ifname> speed <speed>                                             # Фиксированная скорость линка в Mbps (10/100/1000/10000/25000/40000/100000)
set interfaces <ifname> duplex full|half                                          # Дуплекс при отключённом auto-neg
set interfaces <ifname> auto-negotiate on|off                                     # Авто-согласование скорости/дуплекса
set interfaces <ifname> fec rs|fc|none                                            # Forward Error Correction (RS / FireCode / off)
set interfaces <ifname> media-type auto|mdi|mdix                                  # Auto-MDI/MDIX кроссовер (auto=авто, mdi=прямой, mdix=кросс; ethtool -s mdix)
set interfaces <ifname> igmp-snooping fast-leave                                  # Per-port IGMP-snooping fast-leave на мосту (bridge link set fastleave on)
set interfaces <ifname> flow-control rx|tx|both|none                              # 802.3x pause (legacy сахар; раскладывается в pause RX/TX/autoneg)
set interfaces <ifname> pause rx|tx|autoneg on|off                                # Pause-фреймы раздельно по направлению (для RoCEv2 lossless)
set interfaces <ifname> transceiver power-mode-policy high|auto                   # Политика питания QSFP-DD (high для DR8/FR8, auto = умолчание)
set interfaces <ifname> wake-on-lan d|p|u|m|b|a|g|s                               # Wake-on-LAN флаги (как в ethtool -s wol)
set interfaces <ifname> eee on|off                                                # Energy Efficient Ethernet (802.3az)
set interfaces <ifname> offload tso|gso|gro|lro|rx|tx|sg|ntuple|rxhash|highdma|rx-vlan|tx-vlan|hw-tc-offload|l2-fwd-offload on|off  # Включить/отключить конкретный NIC-оффлоад
set interfaces <ifname> ring-buffer rx|tx <size>                                  # Размер кольца дескрипторов RX/TX (frames)
set interfaces <ifname> channels rx|tx|combined|other <count>                     # Количество очередей (channels) NIC
set interfaces <ifname> coalesce adaptive-rx|adaptive-tx on|off                   # Адаптивное coalesce прерываний
set interfaces <ifname> coalesce rx-usecs|tx-usecs|rx-frames|tx-frames|rx-usecs-high|tx-usecs-high <value>  # Параметры coalesce (μs или frames)
set interfaces <ifname> type dummy create                                         # Создать виртуальный dummy-интерфейс
set interfaces <ifname> type loopback create                                      # Создать дополнительный loopback-интерфейс
set interfaces <ifname> bridge <brname>                                           # Привязать порт к bridge как master
set interfaces <ifname> bridge-port bpdu-guard enable|disable                     # BPDU Guard на access-порту
set interfaces <ifname> bridge-port bpdu-filter enable|disable                    # BPDU Filtering (drop tx+rx BPDU; mstpctl setbpdufilter)
delete interfaces <ifname> bridge-port bpdu-filter                                # Отключить BPDU Filtering
set interfaces <ifname> bridge-port cost <value>                                  # STP path-cost
set interfaces <ifname> bridge-port flood on|off                                  # Флудинг unknown-unicast в порт
set interfaces <ifname> bridge-port guard on|off                                  # Root-Guard на bridge-порту
set interfaces <ifname> bridge-port hairpin on|off                                # Hairpin: разрешить TX обратно через входной порт
set interfaces <ifname> bridge-port isolated on|off                               # Port isolation внутри bridge
set interfaces <ifname> bridge-port learning on|off                               # MAC-обучение на порту
set interfaces <ifname> bridge-port mcast-flood on|off                            # Флудинг неизвестного multicast
set interfaces <ifname> bridge-port neigh-suppress on|off                         # ARP/ND suppression (используется EVPN L2VNI)
set interfaces <ifname> bridge-port priority <value>                              # STP port-priority (0–240, шаг 16)
set interfaces <ifname> l2 mode access vlan <vlan_id>                             # Режим access + PVID
set interfaces <ifname> l2 mode trunk allowed-vlans <vlans>                       # Trunk + список тегированных VLAN (диапазоны/csv)
set interfaces <ifname> l2 native-vlan <vlan_id>                                  # Native (untagged) VLAN на trunk-порту
set interfaces <ifname> acl in|out <aclname>                                      # Привязать ACL по направлению (input/output)
set interfaces <ifname> mirror destination <dst-ifname>                           # Локальный SPAN: зеркалировать весь трафик на локальный порт
set interfaces <ifname> mirror remote <collector-ip>                              # RSPAN: зеркалировать на удалённый коллектор (encap по умолчанию gre/gretap)
set interfaces <ifname> mirror remote <collector-ip> encap gre|erspan             # RSPAN с явной инкапсуляцией (gre=gretap L2-копия, erspan=ERSPAN type II)
set interfaces <ifname> mirror remote <collector-ip> encap gre|erspan vrf <vrf>   # RSPAN с инкапсуляцией и VRF коллектора (туннель в указанный VRF)
set interfaces <ifname> lldp enable [on|off]                                      # Включить LLDP TX+RX на порту
set interfaces <ifname> bgp unnumbered                                            # Маркировать порт как BGP unnumbered (RFC 5549, IPv6 LL next-hop)
set interfaces <ifname> bond-member <bond>                                        # Сделать порт slave-ом бонда
set interfaces <ifname> channel-group id <lag_id> mode <mode>                     # Включить порт в LAG (active/passive/balance-rr/...)
set interfaces <ifname> evpn mh es-id <value>                                     # EVPN-MH Ethernet Segment ID (10-байтный hex)
set interfaces <ifname> evpn mh es-sys-mac <mac>                                  # EVPN-MH ESI system-MAC
set interfaces <ifname> evpn mh es-df-pref <pref>                                 # EVPN-MH DF election preference (0–65535)
set interfaces <ifname> evpn mh uplink                                            # Маркировать порт как uplink в EVPN-MH (split-horizon)

delete interfaces <ifname>                                                        # Удалить интерфейс/всю его конфигурацию
delete interfaces <ifname> ipv4 address                                           # Снять все IPv4 адреса
delete interfaces <ifname> ipv6 address                                           # Снять все IPv6 адреса
delete interfaces <ifname> description                                            # Очистить описание
delete interfaces <ifname> mtu                                                    # Сбросить MTU к default
delete interfaces <ifname> speed                                                  # Сбросить фиксированную скорость
delete interfaces <ifname> duplex                                                 # Сбросить duplex
delete interfaces <ifname> auto-negotiate                                         # Сбросить auto-neg
delete interfaces <ifname> fec                                                    # Отключить FEC override
delete interfaces <ifname> media-type                                             # Сбросить media-type (Auto-MDI/MDIX) к default
delete interfaces <ifname> igmp-snooping fast-leave                               # Отключить per-port IGMP fast-leave (fastleave off)
delete interfaces <ifname> flow-control                                           # Отключить 802.3x pause (legacy)
delete interfaces <ifname> pause                                                  # Сбросить все pause-настройки (rx/tx/autoneg)
delete interfaces <ifname> pause rx|tx|autoneg                                    # Сбросить одно направление pause
delete interfaces <ifname> transceiver power-mode-policy                          # Сбросить QSFP-DD power-mode-policy к default
delete interfaces <ifname> eee                                                    # Сбросить EEE
delete interfaces <ifname> wake-on-lan                                            # Очистить WoL флаги
delete interfaces <ifname> offload tso|gso|gro|lro|rx|tx|sg|ntuple|rxhash|highdma|rx-vlan|tx-vlan|hw-tc-offload|l2-fwd-offload  # Сбросить конкретный оффлоад
delete interfaces <ifname> ring-buffer rx|tx                                      # Сбросить ring-buffer override
delete interfaces <ifname> channels                                               # Сбросить channels override
delete interfaces <ifname> coalesce                                               # Сбросить coalesce
delete interfaces <ifname> bridge                                                 # Отвязать порт от bridge
delete interfaces <ifname> bridge-port                                            # Снять весь bridge-port профиль
delete interfaces <ifname> bridge-port bpdu-guard                                 # Отключить BPDU Guard
delete interfaces <ifname> l2                                                     # Снять access/trunk-настройки
delete interfaces <ifname> acl in|out                                             # Снять привязку ACL по направлению
delete interfaces <ifname> mirror                                                 # Снять локальный SPAN с порта
delete interfaces <ifname> mirror remote                                          # Снять RSPAN-сессию (удалить туннель + tc ingress mirred)
delete interfaces <ifname> lldp enable                                            # Отключить LLDP на порту
delete interfaces <ifname> bgp unnumbered                                         # Снять маркер BGP unnumbered
delete interfaces <ifname> bond-member                                            # Изъять порт из бонда
delete interfaces <ifname> channel-group                                          # Изъять порт из LAG
delete interfaces <ifname> evpn mh                                                # Снять EVPN-MH ESI/DF/uplink
delete interfaces <ifname> type dummy                                             # Удалить dummy-интерфейс
delete interfaces <ifname> type loopback                                          # Удалить loopback-интерфейс

    L2 — Канальный уровень¶

    VLAN / Bridges / STP¶

    set vlans vlan <vlan_id> name <name>                                              # Объявить VLAN с именем (vlan_id 1–4094)
set vlans vlan <vlan_range>                                                       # Объявить диапазон VLAN-ов (например 100-110)
set vlans <vlan_id> name <name>                                                   # Краткая форма: VLAN с именем
set vlans <vlan_range>                                                            # Краткая форма: диапазон VLAN-ов
delete vlans vlan <vlan_id|range>                                                 # Удалить VLAN или диапазон
delete vlans <vlan_id|range>                                                      # Краткая форма удаления VLAN

set bridges bridge <name> create                                                  # Создать Linux-bridge
set bridges bridge <name> mtu <mtu>                                               # MTU bridge-интерфейса (64–9216)
set bridges bridge <name> stp enable|disable                                      # Включить/выключить STP на bridge
set bridges bridge <name> stp mode stp|rstp|mstp|none                             # Режим spanning-tree (none = выключен)
set bridges bridge <name> stp priority <val>                                      # Bridge priority (0–61440, шаг 4096)
set bridges bridge <name> stp hello-time <val>                                    # Hello-time в секундах (1–10, по умолчанию 2)
set bridges bridge <name> stp forward-delay <val>                                 # Forward-delay в секундах (4–30, по умолчанию 15)
set bridges bridge <name> stp max-age <val>                                       # Max-age в секундах (6–40, по умолчанию 20)
set bridges bridge <name> igmp-snooping enable|disable                            # IGMP+MLD multicast snooping (kernel mcast_snooping)
set bridges bridge <name> igmp-snooping querier enable|disable                    # Multicast querier на мосту (kernel mcast_querier)
set bridges bridge <name> mac aging-time <seconds>                                # Время старения MAC (сек; kernel ageing_time, ÷100 с)
set bridges bridge <name> vlan-protocol 802.1q|802.1ad                            # VLAN-протокол моста: 802.1q (C-tag) или 802.1ad (Q-in-Q S-tag)
delete bridges bridge <name> vlan-protocol                                        # Вернуть VLAN-протокол к 802.1q
delete bridges bridge <name> igmp-snooping querier                                # Сбросить multicast querier
delete bridges bridge <name> igmp-snooping                                        # Сбросить IGMP+MLD snooping
delete bridges bridge <name> mac aging-time                                       # Вернуть MAC aging к kernel-default
delete bridges bridge <name>                                                      # Удалить bridge целиком
delete bridges bridge <name> stp                                                  # Сбросить STP-настройки bridge

set bridge <bridge> static-fdb mac <mac> port <port> [dst <ip>] [vlan <vid>] [src_vni <vni>]  # Статическая FDB (dst — VXLAN remote VTEP; src_vni — для SVD HER, vid — tagged)
delete bridge <bridge> static-fdb mac <mac> ...                                   # Удалить статическую FDB-запись

set bridge <bridge> multicast-group <group> port <port> [vid <vid>]               # MDB-запись (group = 224.0.0.0/4 или ff00::/8)
delete bridge <bridge> multicast-group ...                                        # Удалить MDB-запись

set mstp region name <name>                                                       # Имя MST-региона (free-form, до 32 символов)
set mstp region revision <val>                                                    # Revision MST-региона (0–65535)
set mstp instance <id> vlan <vlans>                                               # Привязать VLAN(ы) к MSTI (id 1–63)
set mstp instance <id> bridge <name> priority <val>                               # Bridge priority в MSTI (0–61440, шаг 4096)
delete mstp region                                                                # Сбросить настройки MST-региона
delete mstp instance <id>                                                         # Удалить MSTI

    LAG / Port-Channel¶

    set port-channel id <id> members <list> mode active|passive|balance-rr|...        # Создать LAG/bond с членами + режим (active/passive = LACP)
set port-channel id <id> mac <mac>                                                # Жёстко задать MAC бонда (формат aa:bb:cc:dd:ee:ff)
set port-channel <id> lacp-rate slow|fast                                         # LACPDU-период: slow=30s, fast=1s
set port-channel <id> miimon <val>                                                # MII-poll интервал в ms (типично 100)
set port-channel <id> xmit-hash-policy layer2|layer2+3|layer3+4|...               # Алгоритм хеширования TX по членам бонда
set port-channel <id> updelay <val>                                               # Задержка перед up-членом в ms после link-up
set port-channel <id> downdelay <val>                                             # Задержка перед down-членом в ms после link-down

delete port-channel id <id>                                                       # Удалить LAG целиком
delete port-channel <id> lacp-rate                                                # Сбросить lacp-rate к default
delete port-channel <id> miimon                                                   # Сбросить miimon
delete port-channel <id> xmit-hash-policy                                         # Сбросить hash-policy к default (layer2)
delete port-channel <id> updelay                                                  # Сбросить updelay
delete port-channel <id> downdelay                                                # Сбросить downdelay

    MLAG (clagd-style, schema v12)¶

    set mlag domain <id> peer-link <bond> keepalive <ip>                              # MLAG-домен: ID + peer-link bond + keepalive IP соседа
set mlag system-mac <mac>                                                         # Виртуальный system-MAC пары (общий LACP system-id)
set mlag priority <0-65535>                                                       # Приоритет primary-выбора (меньше = выше)
set mlag backup-ip <ip>                                                           # Резервный keepalive IP (защита split-brain)
set mlag dual-attached-bonds <bond1,bond2,...>                                    # Bonds для peer-link FDB sync (extern_learn); auto из port-channels если не задано

delete mlag domain                                                                # Удалить MLAG-домен (отключить mlagd)
delete mlag system-mac                                                            # Сбросить виртуальный system-MAC
delete mlag priority                                                              # Сбросить приоритет в значение по умолчанию
delete mlag backup-ip                                                             # Удалить резервный keepalive адрес
