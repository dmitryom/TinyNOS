    TinyNOS — Справочник команд CLI¶

    Интерфейсы¶

    set interfaces <ifname> ipv4 address <cidr>                                       # Назначить IPv4 адрес/префикс на интерфейс
set interfaces <ifname> ipv6 address <cidr>                                       # Назначить IPv6 адрес/префикс на интерфейс
set interfaces <ifname> description <text>                                        # Текстовое описание порта (free-form, отображается в show)
set interfaces <ifname> mtu <mtu>                                                 # MTU L2-фрейма (64–9216, по умолчанию 1500)
set interfaces <ifname> admin-state up|down                                       # Административное состояние порта (up = поднять, down = погасить)
set interfaces <ifname> address <mac>                                             # Жёстко задать MAC-адрес (формат aa:bb:cc:dd:ee:ff)
set interfaces <ifname> speed <speed>                                             # Фиксированная скорость линка в Mbps (10/100/1000/10000/25000/40000/100000)
set interfaces <ifname> duplex full|half                                          # Дуплекс при отключённом auto-neg
set interfaces <ifname> auto-negotiate on|off                                     # Авто-согласование скорости/дуплекса
set interfaces <ifname> fec rs|fc|none                                            # Forward Error Correction (RS / FireCode / off)
set interfaces <ifname> media-type auto|mdi|mdix                                  # Auto-MDI/MDIX кроссовер (auto=авто, mdi=прямой, mdix=кросс; ethtool -s mdix)
set interfaces <ifname> igmp-snooping fast-leave                                  # Per-port IGMP-snooping fast-leave на мосту (bridge link set fastleave on)
set interfaces <ifname> flow-control rx|tx|both|none                              # 802.3x pause (legacy сахар; раскладывается в pause RX/TX/autoneg)
set interfaces <ifname> pause rx|tx|autoneg on|off                                # Pause-фреймы раздельно по направлению (для RoCEv2 lossless)
set interfaces <ifname> transceiver power-mode-policy high|auto                   # Политика питания QSFP-DD (high для DR8/FR8, auto = умолчание)
set interfaces <ifname> wake-on-lan d|p|u|m|b|a|g|s                               # Wake-on-LAN флаги (как в ethtool -s wol)
set interfaces <ifname> eee on|off                                                # Energy Efficient Ethernet (802.3az)
set interfaces <ifname> offload tso|gso|gro|lro|rx|tx|sg|ntuple|rxhash|highdma|rx-vlan|tx-vlan|hw-tc-offload|l2-fwd-offload on|off  # Включить/отключить конкретный NIC-оффлоад
set interfaces <ifname> ring-buffer rx|tx <size>                                  # Размер кольца дескрипторов RX/TX (frames)
set interfaces <ifname> channels rx|tx|combined|other <count>                     # Количество очередей (channels) NIC
set interfaces <ifname> coalesce adaptive-rx|adaptive-tx on|off                   # Адаптивное coalesce прерываний
set interfaces <ifname> coalesce rx-usecs|tx-usecs|rx-frames|tx-frames|rx-usecs-high|tx-usecs-high <value>  # Параметры coalesce (μs или frames)
set interfaces <ifname> type dummy create                                         # Создать виртуальный dummy-интерфейс
set interfaces <ifname> type loopback create                                      # Создать дополнительный loopback-интерфейс
set interfaces <ifname> bridge <brname>                                           # Привязать порт к bridge как master
set interfaces <ifname> bridge-port bpdu-guard enable|disable                     # BPDU Guard на access-порту
set interfaces <ifname> bridge-port bpdu-filter enable|disable                    # BPDU Filtering (drop tx+rx BPDU; mstpctl setbpdufilter)
delete interfaces <ifname> bridge-port bpdu-filter                                # Отключить BPDU Filtering
set interfaces <ifname> bridge-port cost <value>                                  # STP path-cost
set interfaces <ifname> bridge-port flood on|off                                  # Флудинг unknown-unicast в порт
set interfaces <ifname> bridge-port guard on|off                                  # Root-Guard на bridge-порту
set interfaces <ifname> bridge-port hairpin on|off                                # Hairpin: разрешить TX обратно через входной порт
set interfaces <ifname> bridge-port isolated on|off                               # Port isolation внутри bridge
set interfaces <ifname> bridge-port learning on|off                               # MAC-обучение на порту
set interfaces <ifname> bridge-port mcast-flood on|off                            # Флудинг неизвестного multicast
set interfaces <ifname> bridge-port neigh-suppress on|off                         # ARP/ND suppression (используется EVPN L2VNI)
set interfaces <ifname> bridge-port priority <value>                              # STP port-priority (0–240, шаг 16)
set interfaces <ifname> l2 mode access vlan <vlan_id>                             # Режим access + PVID
set interfaces <ifname> l2 mode trunk allowed-vlans <vlans>                       # Trunk + список тегированных VLAN (диапазоны/csv)
set interfaces <ifname> l2 native-vlan <vlan_id>                                  # Native (untagged) VLAN на trunk-порту
set interfaces <ifname> acl in|out <aclname>                                      # Привязать ACL по направлению (input/output)
set interfaces <ifname> mirror destination <dst-ifname>                           # Локальный SPAN: зеркалировать весь трафик на локальный порт
set interfaces <ifname> mirror remote <collector-ip>                              # RSPAN: зеркалировать на удалённый коллектор (encap по умолчанию gre/gretap)
set interfaces <ifname> mirror remote <collector-ip> encap gre|erspan             # RSPAN с явной инкапсуляцией (gre=gretap L2-копия, erspan=ERSPAN type II)
set interfaces <ifname> mirror remote <collector-ip> encap gre|erspan vrf <vrf>   # RSPAN с инкапсуляцией и VRF коллектора (туннель в указанный VRF)
set interfaces <ifname> lldp enable [on|off]                                      # Включить LLDP TX+RX на порту
set interfaces <ifname> bgp unnumbered                                            # Маркировать порт как BGP unnumbered (RFC 5549, IPv6 LL next-hop)
set interfaces <ifname> bond-member <bond>                                        # Сделать порт slave-ом бонда
set interfaces <ifname> channel-group id <lag_id> mode <mode>                     # Включить порт в LAG (active/passive/balance-rr/...)
set interfaces <ifname> evpn mh es-id <value>                                     # EVPN-MH Ethernet Segment ID (10-байтный hex)
set interfaces <ifname> evpn mh es-sys-mac <mac>                                  # EVPN-MH ESI system-MAC
set interfaces <ifname> evpn mh es-df-pref <pref>                                 # EVPN-MH DF election preference (0–65535)
set interfaces <ifname> evpn mh uplink                                            # Маркировать порт как uplink в EVPN-MH (split-horizon)

delete interfaces <ifname>                                                        # Удалить интерфейс/всю его конфигурацию

delete interfaces <ifname> ipv6 address                                           # Снять все IPv6 адреса

delete interfaces <ifname> pause                                                  # Сбросить все pause-настройки (rx/tx/autoneg)

delete interfaces <ifname> transceiver power-mode-policy                          # Сбросить QSFP-DD power-mode-policy к default
delete interfaces <ifname> eee                                                    # Сбросить EEE
delete interfaces <ifname> wake-on-lan                                            # Очистить WoL флаги
delete interfaces <ifname> offload tso|gso|gro|lro|rx|tx|sg|ntuple|rxhash|highdma|rx-vlan|tx-vlan|hw-tc-offload|l2-fwd-offload  # Сбросить конкретный оффлоад
delete interfaces <ifname> ring-buffer rx|tx                                      # Сбросить ring-buffer override
delete interfaces <ifname> channels                                               # Сбросить channels override
delete interfaces <ifname> coalesce                                               # Сбросить coalesce

    Статическая маршрутизация¶

    set routing static route <prefix> next-hop <ip>                                   # Статический маршрут к prefix через next-hop IP
set routing static route <prefix> next-hop <ip> distance <distance>               # То же + administrative distance (1–255)
set routing static route <prefix> next-hop <ip> weight <1-255>                    # Вес next-hop в multipath-маршруте (UCMP: пропорция hash-бакетов, kernel weighted ECMP)
set routing static route <prefix> blackhole                                       # Маршрут-«чёрная дыра»: трафик к prefix дропается

delete routing static route <prefix>                                              # Удалить все next-hop'ы и blackhole для prefix
delete routing static route <prefix> next-hop <ip>                                # Удалить конкретный next-hop, остальные сохранить
delete routing static route <prefix> next-hop <ip> weight                         # Сбросить вес (равные доли)
delete routing static route <prefix> blackhole                                    # Снять blackhole-флаг с маршрута

    ECMP¶

    set routing ecmp hash-policy layer3|layer4|layer3+4|inner-layer3                  # Политика хэширования ECMP → sysctl fib_multipath_hash_policy: layer3→0 (только IP), layer3+4→1 (5-tuple), layer4→3+hash_fields=0x34 (только порты+proto), inner-layer3→2 (внутренний заголовок туннеля GRE/IPIP; для transit-узлов)
set routing ecmp resilient-hashing                                                # Включить resilient nexthop groups для статических ECMP (consistent hashing, kernel ≥5.13; минимум пересевших потоков при отказе члена)
set routing ecmp dlb enable|disable                                               # Dynamic Load Balancing (адаптивный ECMP) — switch-global режим на виртуальном ASIC (SAI switch-attr; Phase D для силикона)
delete routing ecmp dlb                                                           # Выключить DLB
set routing ecmp glb enable|disable                                               # Global Load Balancing (fabric-wide адаптивный слой над DLB) — switch-side на виртуальном ASIC (Phase D)
delete routing ecmp glb                                                           # Выключить GLB
set routing ecmp resilient-hashing buckets <n>                                    # Число hash-бакетов resilient-группы (степень двойки 32-4096, default 64; больше = точнее пропорции weighted ECMP)

delete routing ecmp resilient-hashing                                             # Вернуться к классическому multipath-маршруту
delete routing ecmp resilient-hashing buckets                                     # Сбросить число бакетов к default (64), resilient остаётся включён

    BGP¶

    set routing bgp local-as <asn>                                                    # Локальная AS (1–4294967295, поддержка 4-byte ASN)
set routing bgp router-id <ip>                                                    # BGP router-id (IPv4, обычно loopback)
set routing bgp bestpath as-path multipath-relax                                  # eBGP multipath при разных AS_PATH
set routing bgp cluster-id <ip>                                                   # Cluster-id для route-reflector
set routing bgp enforce-first-as enable|disable                                   # Требовать peer-AS первой в AS_PATH eBGP-апдейтов
set routing bgp graceful-restart                                                  # Включить BGP Graceful Restart (RFC 4724)
set routing bgp graceful-restart restart-time <val>                               # Restart-time в секундах (по умолчанию 120)
set routing bgp graceful-restart stalepath-time <seconds>                         # Stale-path retention для GR-helper (1–3600 сек)
set routing bgp graceful-restart preserve-fw-state                                # Preserve forwarding state (RFC 4724 F-bit) — для ISSU
set routing bgp graceful-restart-helper                                           # GR helper-only mode (peer restartable, мы — нет)
set routing bgp maximum-paths <n>                                                 # ECMP-параллелизм для eBGP (1–256)
set routing bgp maximum-paths ibgp <n>                                            # ECMP-параллелизм для iBGP (1–256)
